-- Button für das "Bodenziehen"
local groundButton = Instance.new("TextButton")
groundButton.Size = UDim2.new(1, -10, 0, 40)
groundButton.Position = UDim2.new(0, 5, 0, 210)
groundButton.Text = "Ground Sink"
groundButton.TextColor3 = Color3.fromRGB(255, 255, 255)
groundButton.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
groundButton.Font = Enum.Font.SourceSans
groundButton.TextSize = 18
groundButton.Parent = mainMenu

-- Funktion zum Boden ziehen und wieder hochkommen
groundButton.MouseButton1Click:Connect(function()
    local player = game.Players.LocalPlayer
    local character = player.Character
    local humanoidRootPart = character:FindFirstChild("HumanoidRootPart")
    
    if character and humanoidRootPart then
        -- Spieler nach unten ziehen
        local targetPosition = humanoidRootPart.Position - Vector3.new(0, 5, 0) -- Zieht den Spieler 5 Studs in den Boden
        humanoidRootPart.CFrame = CFrame.new(targetPosition)
        
        -- Spieler bleibt für 3 Sekunden unten
        task.wait(3)
        
        -- Spieler nach oben bringen
        local newPosition = humanoidRootPart.Position + Vector3.new(0, 5, 0) -- Spieler wird um 5 Studs nach oben bewegt
        humanoidRootPart.CFrame = CFrame.new(newPosition)
    end
end)
