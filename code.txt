-- Variablen für Toggles
local zenkaiEnabled = false
local attackEnabled = false
local agilityEnabled = false
local kiEnabled = false

-- Funktionalität
SubmitButton.MouseButton1Click:Connect(function()
    if KeyInput.Text == keyRequired then
        isUnlocked = true
        Frame.Visible = false
        ToggleButton.Visible = true
    else
        KeyInput.Text = "Invalid Key"
    end
end)

ToggleButton.MouseButton1Click:Connect(function()
    Menu.Visible = not Menu.Visible
end)

ZenkaiButton.MouseButton1Click:Connect(function()
    zenkaiEnabled = not zenkaiEnabled
    if zenkaiEnabled then
        ZenkaiButton.Text = "Auto Zenkai Farm: ON"
        print("Zenkai Farm aktiviert!")
        -- Füge hier den Code für Auto Zenkai Farm ein
    else
        ZenkaiButton.Text = "Auto Zenkai Farm: OFF"

        -- Beende hier den Code für Auto Zenkai Farm
    end
end)
